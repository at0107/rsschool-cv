Simple function to calculate the average of array values

                    
    function digPow(n, p) {
    let digits = String(n).split('').map(Number);
    
    let sum = digits.reduce((acc, digit, index) => {
        return acc + digit ** (p + index);
    }, 0);
    
    return sum % n === 0 ? sum / n : -1;
}
